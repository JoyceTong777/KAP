Query Example

Query pods information
select pods.cluster, pods.metadata->name, pods.metadata->namespace, pods.status->podIP, pods.status->phase from pods
Query the nodes information of a specific cluster
select nodes.cluster, nodes.metadata->name from nodes where nodes.cluster like 'Cluster1%'
Query the services information of a specific namespace in multiple clusters
select services.cluster, services.metadata->name, services.metadata->namespace, services.spec->clusterIP from services where services.metadata->namespace="default"
